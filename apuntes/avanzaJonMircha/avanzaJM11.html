       function cuadrado(valor){
        setTimeout(()=>{
            return console.log({
            valor,
            resultado:valor*valor
        })
        },Math.random() * 1000);
        
       }
       function* generador(){
        console.log("Inicia Generador");
       yield cuadrado(0)
       yield  cuadrado(1)
       yield  cuadrado(2)
       yield cuadrado(3)
       yield  cuadrado(4)
       yield  cuadrado(5)
        console.log("Termina Generador");
       }
       let gen = generador();
       for(let y of gen){
        console.log(y);
       }
 </script>
