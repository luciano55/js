Temporizadores (setTimeout & setInterval)

  • Video 44
<script>
        console.log("Inicio");
        setTimeout(()=>{
        console.log("Ejecutando un setTimeout, esto se ejecuta una sóla vez");

        },1000);

        setInterval(()=>{
        console.log("Ejecutando un setInterval, indefinidamente cada segundo");

        },1000);

        setInterval(()=>{
        console.log(new Date().toLocaleTimeString());

        },1000);

        let temporizador = setTimeout(()=>{
        console.log("Ejecutando un setTimeout, esto se ejecuta una sóla vez");

        },1000);
        clearTimeout(temporizador);

        let temporizadorotro = setInterval(()=>{
        console.log(new Date().toLocaleTimeString());

        },1000);

        clearInterval(temporizadorotro);
  </script>
