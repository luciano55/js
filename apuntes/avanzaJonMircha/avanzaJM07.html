El objeto Set permite almacenar valores únicos de cualquier tipo, incluso valores primitivos u referencias a objetos.

<script>
    const set = new Set([1,2,3,3,4,true,false,true,{},{},"Hola","HOLA"]); 
    console.log(set);
    console.log(set.size);
    const set2 = new Set(); 
    set2.add(1);
    set2.add(2);
    set2.add(3);
    set2.add(3);
    set2.add(4);
    set2.add(true);
    set2.add(false);
    set2.add(true);
    set2.add({});
    set2.add({});
    set2.add("hola");
    set2.add("HOLA");
    console.log(set2);
    console.log(set2.size);
    console.log("Recorriendo set");
    for(item of set){
        console.log(item);
    }
    console.log("Recorriendo set2");
    set2.forEach(item => {
        console.log(item);
    });
    // set es una colección NO un array
    console.log(set[0]); // undefined
    console.log(Array.from(set)[0]);  // 1  Ahora si
 let arr = Array.from(set);
 console.log(arr);
 console.log(arr[0]);
 console.log(arr[4]);

 set.delete("HOLA");
 console.log(set);
 console.log(set.has("Hola")); // true
 console.log(set.has({})); // false

 set2.clear();
 console.log(set2);
  </script>
