// Min 1 12) Programa una función que determine si un número es primo 
//(aquel que solo es divisible por sí mismo y 1) o no, pe. miFuncion(7) devolverá true.

const numeroPrimo = (numero = undefined)=>{
      if(numero === undefined ) return console.error("NO es un número");
      if(numero === 0) return console.error("El número no puede ser 0");
      if(numero === 1) return console.error("El número no puede ser 1");
      if(Math.sign(numero)=== -1) return console.error("El número no puede ser negativo");
       let divisible = false;
       for(let i = 2; i < numero; i++){
           if(numero % i) {
               divisible = true;
               break;
           }
       }
       return (!divisible) 
        ? console.info(`El numero ${numero}, NO es primo`)
       :console.info(`El numero ${numero}, es primo`)
  }
  numeroPrimo();
  numeroPrimo(1);
  numeroPrimo(7);

 // Min:12 13) Programa una función que determine si un número es par 

const numeroParImpar = (numero = undefined)=>{
    if(numero === undefined ) return console.error("NO es un número");
    if(typeof numero !== "number") return console.error(`El número ${numero} ingresado No es un número`);
    return (numero % 2 === 0) 
    ? console.info(`El numero ${numero}, es Par`)
    :console.info(`El numero ${numero}, es Impar`)
}

numeroParImpar("Hola");
numeroParImpar();
numeroParImpar(3);
numeroParImpar(22);
numeroParImpar(-322);

// Min:16 14) Programa una función para convertir grados Celsius a Fahrenhei
const convertirGrado = (grados = undefined, unidad = undefined)=>{
    if(grados === undefined) return console.warn("No ingresaste grados a convertir");
    if(typeof grados !== 'number') return console.error(`El valor "${grados}" ingresado, NO es un Número`);
    if(unidad === undefined) return console.warn("No ingresaste tipo de grados a convertir");
    if(typeof unidad !== 'string') return console.error(`El valor "${unidad}" ingresado, NO es una cadena de texto`);
   // if(unidad.length !== 1 || unidad !== 'c' || unidad !== 'f') return console.error(`El valor "${unidad}" ingresado, NO es una unidad requerida`);
    if(unidad.length !== 1 || !/(C|F)/.test(unidad))  return console.error(`El valor "${unidad}" ingresado, NO es una unidad requerida`);

    if(unidad === 'C'){
        return console.info(`${grados}ºC = ${Math.round(grados * (9/5)+ 32)}ºF`);
    }else  if(unidad === 'F'){
        return console.info(`${grados}ºF = ${Math.round((grados - 32)  * (5/9))}ºC`);
    }
}
convertirGrado(1,"W");
convertirGrado(100,"C");
convertirGrado(32,"F");
convertirGrado(100,"F");
